<a name="kode-lengkap"></a>
## ğŸ“¦ Kode Lengkap

Berikut adalah kode lengkap yang sudah kita buat:

```javascript
// ========================================
// VARIABEL GLOBAL
// ========================================
const itemForm = document.getElementById('item-form');
const itemInput = document.getElementById('item-input');
const itemList = document.getElementById('item-list');
const clearBtn = document.getElementById('clear');

// ========================================
// FUNGSI: TAMBAH ITEM
// ========================================
function addItem(e) {
  e.preventDefault();

  const newItem = itemInput.value;

  // Validasi Input
  if (newItem === '') {
    alert('Please add an item');
    return;
  }

  // Buat list item
  const li = document.createElement('li');
  li.appendChild(document.createTextNode(newItem));

  const button = createButton('remove-item btn-link text-red');
  li.appendChild(button);

  itemList.appendChild(li);

  itemInput.value = '';
}

// ========================================
// FUNGSI: BUAT BUTTON
// ========================================
function createButton(classes) {
  const button = document.createElement('button');
  button.className = classes;
  const icon = createIcon('fa-solid fa-xmark');
  button.appendChild(icon);
  return button;
}

// ========================================
// FUNGSI: BUAT ICON
// ========================================
function createIcon(classes) {
  const icon = document.createElement('i');
  icon.className = classes;
  return icon;
}

// ========================================
// FUNGSI: HAPUS ITEM (Event Delegation)
// ========================================
function removeItem(e) {
  // Cek apakah yang diklik adalah tombol remove
  if (e.target.parentElement.classList.contains('remove-item')) {
    // Hapus list item (naik 2 level ke parent)
    e.target.parentElement.parentElement.remove();
  }
}

// ========================================
// FUNGSI: HAPUS SEMUA ITEM
// ========================================
function clearItems() {
  // Metode While Loop (Recommended)
  while (itemList.firstChild) {
    itemList.removeChild(itemList.firstChild);
  }
}

// ========================================
// EVENT LISTENERS
// ========================================
itemForm.addEventListener('submit', addItem);
itemList.addEventListener('click', removeItem);
clearBtn.addEventListener('click', clearItems);
```
