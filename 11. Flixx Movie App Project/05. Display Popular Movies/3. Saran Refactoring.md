const global = {
  currentPage: window.location.pathname,
};

// ==============================
// DISPLAY POPULAR MOVIES
// ==============================

const IMAGE_BASE_URL = 'https://image.tmdb.org/t/p/w500';
const popularMoviesContainer = document.querySelector('#popular-movies');

async function displayPopularMovies() {
  const data = await fetchAPIData('movie/popular');
  const movies = data.results;

  movies.forEach(createMovieCard);
}

function createMovieCard(movie) {
  const movieCard = document.createElement('div');
  movieCard.classList.add('card');

  movieCard.innerHTML = `
    <a href="movie-details.html?id=${movie.id}">
      <img
        src="${getMoviePoster(movie.poster_path)}"
        class="card-img-top"
        alt="${movie.title}"
      />
    </a>
    <div class="card-body">
      <h5 class="card-title">${movie.title}</h5>
      <p class="card-text">
        <small class="text-muted">Release: ${movie.release_date}</small>
      </p>
    </div>
  `;

  popularMoviesContainer.appendChild(movieCard);
}

function getMoviePoster(posterPath) {
  if (!posterPath) {
    return 'images/no-image.jpg';
  }

  return `${IMAGE_BASE_URL}${posterPath}`;
}

// ==============================
// FETCH DATA FROM TMDB API
// ==============================

async function fetchAPIData(endpoint) {
  const API_KEY = '';
  const API_URL = 'https://api.themoviedb.org/3/';

  const response = await fetch(
    `${API_URL}${endpoint}?api_key=${API_KEY}&language=en-US`
  );

  return response.json();
}

// ==============================
// HIGHLIGHT ACTIVE LINK
// ==============================

function highlightActiveLink() {
  const links = document.querySelectorAll('.nav-link');

  links.forEach((link) => {
    if (link.getAttribute('href') === global.currentPage) {
      link.classList.add('active');
    }
  });
}

// ==============================
// INIT APP
// ==============================

function init() {
  switch (global.currentPage) {
    case '/':
    case '/index.html':
      displayPopularMovies();
      break;
    case '/shows.html':
      console.log('Shows');
      break;
    case '/movie-details.html':
      console.log('Movie Details');
      break;
    case '/tv-details.html':
      console.log('TV Details');
      break;
    case '/search.html':
      console.log('Search');
      break;
  }

  highlightActiveLink();
}

document.addEventListener('DOMContentLoaded', init);

### Claude AI

```javascript
// ========== KONFIGURASI ==========
const CONFIG = {
  API_KEY: '', // Isi dengan API key dari TMDB
  API_BASE_URL: 'https://api.themoviedb.org/3/',
  IMAGE_BASE_URL: 'https://image.tmdb.org/t/p/w500',
  NO_IMAGE_PATH: '../images/no-image.jpg'
};

const PAGES = {
  HOME: '/',
  INDEX: '/index.html',
  SHOWS: '/shows.html',
  MOVIE_DETAILS: '/movie-details.html',
  TV_DETAILS: '/tv-details.html',
  SEARCH: '/search.html'
};

// ========== STATE MANAGEMENT ==========
const appState = {
  currentPage: window.location.pathname
};

// ========== API FUNCTIONS ==========
/**
 * Mengambil data dari TMDB API
 * @param {string} endpoint - Endpoint API yang akan diakses
 * @returns {Promise<Object>} Data dari API
 */
async function fetchMovieData(endpoint) {
  try {
    const url = `${CONFIG.API_BASE_URL}${endpoint}?api_key=${CONFIG.API_KEY}&language=en-US`;
    const response = await fetch(url);
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    const data = await response.json();
    return data;
  } catch (error) {
    console.error('Error fetching data:', error);
    throw error;
  }
}

/**
 * Mengambil daftar film populer
 * @returns {Promise<Array>} Array berisi daftar film populer
 */
async function getPopularMovies() {
  const data = await fetchMovieData('movie/popular');
  return data.results;
}

// ========== UI FUNCTIONS ==========
/**
 * Membuat URL gambar poster film
 * @param {string|null} posterPath - Path poster dari API
 * @returns {string} URL lengkap gambar poster
 */
function createPosterImageUrl(posterPath) {
  if (posterPath) {
    return `${CONFIG.IMAGE_BASE_URL}${posterPath}`;
  }
  return CONFIG.NO_IMAGE_PATH;
}

/**
 * Membuat elemen HTML untuk card film
 * @param {Object} movie - Data film dari API
 * @returns {HTMLElement} Element div berisi card film
 */
function createMovieCard(movie) {
  const cardElement = document.createElement('div');
  cardElement.classList.add('card');
  
  const posterUrl = createPosterImageUrl(movie.poster_path);
  
  cardElement.innerHTML = `
    <a href="movie-details.html?id=${movie.id}">
      <img
        src="${posterUrl}"
        class="card-img-top"
        alt="${movie.title}"
      />
    </a>
    <div class="card-body">
      <h5 class="card-title">${movie.title}</h5>
      <p class="card-text">
        <small class="text-muted">Release: ${movie.release_date}</small>
      </p>
    </div>
  `;
  
  return cardElement;
}

/**
 * Menampilkan daftar film populer ke halaman
 */
async function displayPopularMovies() {
  try {
    const movies = await getPopularMovies();
    const movieContainer = document.querySelector('#popular-movies');
    
    // Bersihkan container terlebih dahulu
    movieContainer.innerHTML = '';
    
    // Render setiap film
    movies.forEach((movie) => {
      const movieCard = createMovieCard(movie);
      movieContainer.appendChild(movieCard);
    });
  } catch (error) {
    console.error('Error displaying movies:', error);
    showErrorMessage('Gagal memuat daftar film. Silakan coba lagi.');
  }
}

/**
 * Menampilkan pesan error ke user
 * @param {string} message - Pesan error yang akan ditampilkan
 */
function showErrorMessage(message) {
  const movieContainer = document.querySelector('#popular-movies');
  movieContainer.innerHTML = `
    <div class="error-message">
      <p>${message}</p>
    </div>
  `;
}

// ========== NAVIGATION ==========
/**
 * Highlight link navigasi yang sedang aktif
 */
function highlightActiveNavLink() {
  const navLinks = document.querySelectorAll('.nav-link');
  
  navLinks.forEach((link) => {
    const linkHref = link.getAttribute('href');
    
    if (linkHref === appState.currentPage) {
      link.classList.add('active');
    }
  });
}

// ========== ROUTER ==========
/**
 * Routing halaman berdasarkan URL
 */
function routePage() {
  const currentPage = appState.currentPage;
  
  switch (currentPage) {
    case PAGES.HOME:
    case PAGES.INDEX:
      displayPopularMovies();
      break;
      
    case PAGES.SHOWS:
      console.log('Shows page');
      break;
      
    case PAGES.MOVIE_DETAILS:
      console.log('Movie Details page');
      break;
      
    case PAGES.TV_DETAILS:
      console.log('TV Details page');
      break;
      
    case PAGES.SEARCH:
      console.log('Search page');
      break;
      
    default:
      console.log('Page not found');
  }
}

// ========== INITIALIZATION ==========
/**
 * Inisialisasi aplikasi
 */
function initializeApp() {
  routePage();
  highlightActiveNavLink();
}

// Jalankan aplikasi setelah DOM selesai dimuat
document.addEventListener('DOMContentLoaded', initializeApp);
```
