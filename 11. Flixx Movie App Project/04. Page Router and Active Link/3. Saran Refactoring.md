### Versi Chat GPT

```javascript
// Menyimpan state global aplikasi
// currentPath berisi path URL halaman yang sedang aktif
const appState = {
  currentPath: window.location.pathname,
};

// Menandai link navigasi yang sesuai dengan halaman aktif
function highlightActiveLink(currentPath) {
  document.querySelectorAll('.nav-link').forEach((link) => {
    // Ambil pathname dari href link
    const linkPath = new URL(link.href).pathname;

    // Tambahkan atau hapus class 'active' sesuai halaman aktif
    link.classList.toggle('active', linkPath === currentPath);
  });
}

// Menjalankan logic sesuai halaman yang sedang dibuka
function handlePageLoad(currentPath) {
  // Mapping path ke aksi halaman
  const pageActions = {
    '/': () => console.log('Home'),
    '/index.html': () => console.log('Home'),
    '/shows.html': () => console.log('Shows'),
    '/movie-details.html': () => console.log('Movie Details'),
    '/tv-details.html': () => console.log('TV Details'),
    '/search.html': () => console.log('Search'),
  };

  // Jalankan aksi jika path ditemukan
  pageActions[currentPath]?.();
}

// Entry point aplikasi
function init() {
  const { currentPath } = appState;

  // Jalankan logic halaman
  handlePageLoad(currentPath);

  // Highlight menu navigasi yang aktif
  highlightActiveLink(currentPath);
}

// Menjalankan aplikasi setelah DOM selesai dimuat
document.addEventListener('DOMContentLoaded', init);

```

```javascript
// Configuration
const APP_CONFIG = {
  currentPath: window.location.pathname,
  routes: {
    HOME: ['/', '/index.html'],
    SHOWS: '/shows.html',
    MOVIE_DETAILS: '/movie-details.html',
    TV_DETAILS: '/tv-details.html',
    SEARCH: '/search.html',
  },
};

// DOM Selectors
const DOM_SELECTORS = {
  navLink: '.nav-link',
  activeClass: 'active',
};

/**
 * Highlights the active navigation link based on current page
 */
function setActiveNavigationLink() {
  const navLinks = document.querySelectorAll(DOM_SELECTORS.navLink);
  
  navLinks.forEach((link) => {
    const href = link.getAttribute('href');
    
    if (href === APP_CONFIG.currentPath) {
      link.classList.add(DOM_SELECTORS.activeClass);
    } else {
      link.classList.remove(DOM_SELECTORS.activeClass);
    }
  });
}

/**
 * Handles page-specific initialization logic
 * @param {string} currentPath - The current page path
 */
function handlePageRouting(currentPath) {
  const { routes } = APP_CONFIG;
  
  switch (currentPath) {
    case routes.HOME[0]:
    case routes.HOME[1]:
      initHomePage();
      break;
    case routes.SHOWS:
      initShowsPage();
      break;
    case routes.MOVIE_DETAILS:
      initMovieDetailsPage();
      break;
    case routes.TV_DETAILS:
      initTVDetailsPage();
      break;
    case routes.SEARCH:
      initSearchPage();
      break;
    default:
      handleUnknownRoute(currentPath);
  }
}

// Page initialization functions
function initHomePage() {
  console.log('[Router] Home page loaded');
  // Add home page specific logic here
}

function initShowsPage() {
  console.log('[Router] Shows page loaded');
  // Add shows page specific logic here
}

function initMovieDetailsPage() {
  console.log('[Router] Movie Details page loaded');
  // Add movie details specific logic here
}

function initTVDetailsPage() {
  console.log('[Router] TV Details page loaded');
  // Add TV details specific logic here
}

function initSearchPage() {
  console.log('[Router] Search page loaded');
  // Add search page specific logic here
}

function handleUnknownRoute(path) {
  console.warn(`[Router] Unknown route: ${path}`);
  // Optionally redirect to 404 page
}

/**
 * Main application initialization
 */
function initializeApp() {
  try {
    handlePageRouting(APP_CONFIG.currentPath);
    setActiveNavigationLink();
  } catch (error) {
    console.error('[App] Initialization error:', error);
  }
}

// Start application when DOM is ready
document.addEventListener('DOMContentLoaded', initializeApp);
```
