// Menyimpan informasi global aplikasi
const appState = {
  currentPage: window.location.pathname,

// Konfigurasi TMDB API
const TMDB_CONFIG = {

  getElement('.spinner').classList.remove('show');

// Fetch data dari TMDB API
const fetchFromTMDB = async (endpoint) => {
  const response = await fetch(
    `${TMDB_CONFIG.BASE_URL}${endpoint}?api_key=${TMDB_CONFIG.API_KEY}&language=en-US`
  );

  return await response.json();
};

/* =====================================================
   CARD CREATORS
===================================================== */

// Membuat card Movie
const createMovieCard = (movie) => {
  const card = document.createElement('div');
  card.classList.add('card');

  const posterImage = movie.poster_path
    ? `${TMDB_CONFIG.IMAGE_BASE_URL}${movie.poster_path}`
    : '../images/no-image.jpg';

  card.innerHTML = `
    <a href="movie-details.html?id=${movie.id}">
      <img src="${posterImage}" class="card-img-top" alt="${movie.title}" />
    </a>
    <div class="card-body">
      <h5 class="card-title">${movie.title}</h5>

  const card = document.createElement('div');

    ? `${TMDB_CONFIG.IMAGE_BASE_URL}${tvShow.poster_path}`
    : '../images/no-image.jpg';

  card.innerHTML = `
    <a href="tv-details.html?id=${tvShow.id}">
      <img src="${posterImage}" class="card-img-top" alt="${tvShow.name}" />

    <div class="card-body">
      <h5 class="card-title">${tvShow.name}</h5>
      <p class="card-text">

      </p>
    </div>
  `;

  return card;

   üé¨ MOVIE DETAILS HELPERS (HIGHLIGHTED)

const getQueryParam = (key) => {

  return params.get(key);
};

// üî• Format angka (budget, revenue)
const formatCurrency = (number) => {
  return number.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ',');
};

// üî• Poster movie dengan fallback
const getMoviePoster = (posterPath, title) => {
  return posterPath
    ? `<img src="${TMDB_CONFIG.IMAGE_BASE_URL}${posterPath}" alt="${title}" />`

  const movieDetails = createMovieDetails(movie);

  getElement('#movie-details').appendChild(movieDetails);

  hideLoadingSpinner();

   NAVIGATION

    if (link.getAttribute('href') === appState.currentPage) {
      link.classList.add('active');
    }
  });

   APP INITIALIZER (ENTRY POINT)

  switch (appState.currentPage) {
    case '/':
    case '/index.html':
      loadPopularMovies();
      break;

    case '/shows.html':
      loadPopularTvShows();
      break;

    case '/movie-details.html':
      loadMovieDetails(); // üé¨üî•
      break;
  }

  highlightActiveNavigation();
};

// Jalankan aplikasi
document.addEventListener('DOMContentLoaded', initializeApp);
