### Kode JavaScript

```javascript
function clock() {
  const now = new Date();
  const canvas = document.getElementById('canvas');
  const ctx = canvas.getContext('2d');
  // Setup canvas
  ctx.save();
  ctx.clearRect(0, 0, 500, 500);
  ctx.translate(250, 250);
  ctx.rotate(-Math.PI / 2);
  // Set default styles
  ctx.strokeStyle = '#000000';
  ctx.fillStyle = '#ffffff'; // ðŸŽ¨ Background putih bersih
  ctx.lineWidth = 5;
  ctx.lineCap = 'round';
  
  // Draw clock face/border
  ctx.save();
  ctx.beginPath();
  ctx.lineWidth = 14;
  ctx.strokeStyle = '#800000';
  ctx.arc(0, 0, 142, 0, Math.PI * 2, true);
  ctx.stroke();
  ctx.fill();
  ctx.restore();
  
  // ðŸ”µ Draw minute dots (titik-titik dengan ukuran berbeda) ðŸ”µ
  ctx.save();
  ctx.fillStyle = '#333333'; // ðŸŽ¨ Warna abu gelap untuk dots
  for (let i = 0; i < 60; i++) {
    ctx.beginPath();
    // ðŸŽ¯ Titik kelipatan 5 (posisi jam) lebih besar â­•
    if (i % 5 === 0) {
      ctx.arc(125, 0, 4, 0, Math.PI * 2, true); // â­• Titik BESAR radius 4px
    } else {
      ctx.arc(125, 0, 2, 0, Math.PI * 2, true); // âš« Titik kecil radius 2px
    }
    ctx.fill();
    ctx.rotate(Math.PI / 30);
  }
  ctx.restore();
  
  // âœ¨ Draw hour numbers (1-12) dengan font tipis âœ¨
  ctx.save();
  ctx.font = '32px Arial'; // ðŸ“ Font lebih besar & tipis (tanpa bold)
  ctx.fillStyle = '#333333'; // ðŸŽ¨ Warna abu gelap
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  
  for (let i = 1; i <= 12; i++) {
    const angle = (Math.PI / 6) * i;
    const x = Math.cos(angle) * 95; // ðŸ“ Posisi lebih keluar (95px)
    const y = Math.sin(angle) * 95; // ðŸ“ Posisi lebih keluar (95px)
    
    ctx.save();
    ctx.translate(x, y);
    ctx.rotate(Math.PI / 2);
    ctx.fillText(i.toString(), 0, 0);
    ctx.restore();
  }
  ctx.restore();
  
  // ðŸ“ Draw "Edison" text (atau bisa diganti nama lain) ðŸ“
  ctx.save();
  ctx.rotate(Math.PI / 2);
  ctx.font = '18px serif'; // ðŸŽ¨ Font serif untuk kesan elegan
  ctx.fillStyle = '#555555'; // ðŸŽ¨ Warna abu medium
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText('Edison', 0, -40); // ðŸ“ Posisi di atas center
  ctx.restore();
  
  // Get current time
  const hr = now.getHours() % 12;
  const min = now.getMinutes();
  const sec = now.getSeconds();
  
  // ðŸ• Draw hour hand (jarum jam hitam tebal) ðŸ•
  ctx.save();
  ctx.rotate(
    (Math.PI / 6) * hr + (Math.PI / 360) * min + (Math.PI / 21600) * sec
  );
  ctx.strokeStyle = '#000000'; // âš« Hitam
  ctx.lineWidth = 8; // ðŸ“ Agak tebal
  ctx.lineCap = 'round'; // ðŸ”µ Ujung bulat
  ctx.beginPath();
  ctx.moveTo(-15, 0);
  ctx.lineTo(60, 0); // ðŸ“ Lebih pendek
  ctx.stroke();
  ctx.restore();
  
  // â° Draw minute hand (jarum menit hitam tipis) â°
  ctx.save();
  ctx.rotate((Math.PI / 30) * min + (Math.PI / 1800) * sec);
  ctx.strokeStyle = '#000000'; // âš« Hitam
  ctx.lineWidth = 5; // ðŸ“ Lebih tipis dari jarum jam
  ctx.lineCap = 'round'; // ðŸ”µ Ujung bulat
  ctx.beginPath();
  ctx.moveTo(-20, 0);
  ctx.lineTo(95, 0); // ðŸ“ Lebih panjang
  ctx.stroke();
  ctx.restore();
  
  // â±ï¸ Draw second hand (jarum detik merah tipis) â±ï¸
  ctx.save();
  ctx.rotate((sec * Math.PI) / 30);
  ctx.strokeStyle = '#FF0000'; // ðŸ”´ Merah terang
  ctx.lineWidth = 2; // ðŸ“ Sangat tipis
  ctx.lineCap = 'round'; // ðŸ”µ Ujung bulat
  ctx.beginPath();
  ctx.moveTo(-25, 0);
  ctx.lineTo(110, 0); // ðŸ“ Paling panjang
  ctx.stroke();
  
  // â­• Draw center circles (lingkaran tengah) â­•
  ctx.beginPath();
  ctx.arc(0, 0, 8, 0, Math.PI * 2, true); // ðŸ”´ Lingkaran luar merah
  ctx.fillStyle = '#FF0000';
  ctx.fill();
  
  ctx.beginPath();
  ctx.arc(0, 0, 4, 0, Math.PI * 2, true); // âšª Lingkaran dalam putih
  ctx.fillStyle = '#ffffff';
  ctx.fill();
  
  ctx.restore();
  
  // ðŸ• Draw Digital Clock ðŸ•
  ctx.save();
  ctx.rotate(Math.PI / 2);
  
  // ðŸ“¦ Draw background box untuk digital clock
  ctx.fillStyle = '#800000';
  ctx.fillRect(-80, 170, 160, 45);
  
  // â° Format waktu digital
  const hours = now.getHours().toString().padStart(2, '0');
  const minutes = now.getMinutes().toString().padStart(2, '0');
  const seconds = now.getSeconds().toString().padStart(2, '0');
  const digitalTime = `${hours}:${minutes}:${seconds}`;
  
  // âœï¸ Draw digital time text
  ctx.font = 'bold 28px monospace';
  ctx.fillStyle = '#f4f4f4';
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText(digitalTime, 0, 192);
  
  ctx.restore();
  // ðŸ• End of Digital Clock ðŸ•
  
  ctx.restore();
  requestAnimationFrame(clock);
}
requestAnimationFrame(clock);
```
