const SpeechRecognition = 
  window.SpeechRecognition || window.webkitSpeechRecognition;

// Inisialisasi Speech Recognition object
const rec = new SpeechRecognition();

// Konfigurasi
rec.lang = 'en-US';        // Bahasa: English (US)
rec.continuous = true;     // Terus mendengarkan

// Event handler ketika ada hasil recognition
rec.onresult = function(e) {
  // Array warna yang diterima
  const acceptedColors = [
    'red',
    'blue',
    'green',
    'yellow',
    'pink',
    'brown',
    'purple',
    'orange',
    'black',
    'white'
  ];

  // Loop semua hasil baru
  for (let i = e.resultIndex; i < e.results.length; i++) {
    // Ambil transcript, lowercase, dan trim
    const script = e.results[i][0].transcript
      .toLowerCase()
      .trim();
    
    // Validasi dan ubah background
    if (acceptedColors.includes(script)) {
      document.body.style.backgroundColor = script;
    } else {
      alert('Please say a color');
    }
  }
};

// Mulai listening!
rec.start();
