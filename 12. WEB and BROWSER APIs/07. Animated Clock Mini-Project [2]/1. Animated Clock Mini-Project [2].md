```javascript
function clock() {

  ctx.fillStyle = '#f4f4f4';
  ctx.lineWidth = 5;
  ctx.lineCap = 'round';

  // ... (kode untuk menggambar akan ditambah di sini)

  ctx.restore(); // Kembalikan state
  requestAnimationFrame(clock); // Panggil function ini lagi

// Jalankan function pertama kali

```

**Penjelasan Penting:**

```javascript

```
Ini adalah "kuas cat" kita. Dengan `ctx`, kita bisa:
- Menggambar garis: `ctx.stroke()`
- Mengisi warna: `ctx.fill()`
- Membuat kurva: `ctx.arc()`

Ini memindahkan titik awal (0,0) dari pojok kiri atas ke tengah canvas.

## ğŸŒˆ Mengubah Warna Dinamis dengan Input

Sekarang kita akan "hubungkan" input warna dengan gambar jam. Kita perlu mengubah nilai warna hardcoded menjadi mengambil nilai dari input.

### Sebelum (Warna Hardcoded):


```javascript

ctx.strokeStyle = borderColor.value; // Ambil dari input
```

### Contoh Lengkap: Menggambar Face (Lingkaran Jam)

```javascript
// SESUDAH (DINAMIS):

ctx.fillStyle = faceColor.value;      // â† Ambil dari input

ctx.stroke();
ctx.fill();
ctx.restore();
```

### Mengubah Warna Number Lines (Angka di Jam):

```javascript
// Draw hour lines

```

<a name="save-image"></a>
## ğŸ’¾ Menyimpan Jam Analog sebagai Gambar

### Konsep: Mengubah Canvas menjadi Gambar

Canvas bisa diubah menjadi file gambar PNG. Caranya:

```
Canvas â†’ URL Data â†’ Buat Link â†’ Download File
```

### Kode Lengkap:

```javascript
// Ambil button save
document.getElementById('save-btn').addEventListener('click', () => {
  // STEP 1: Ubah canvas menjadi data URL (gambar)
  const dataURL = canvas.toDataURL('image/png');
  
  // STEP 2: Buat element link
  const link = document.createElement('a');
  
  // STEP 3: Atur nama file yang akan didownload
  link.download = 'clock.png';
  
  // STEP 4: Atur URL untuk di-download
  link.href = dataURL;
  
  // STEP 5: Klik link (triggering download)
  link.click();
});
```

### Penjelasan Detail:

#### STEP 1: `canvas.toDataURL('image/png')`

```javascript
const dataURL = canvas.toDataURL('image/png');
```

Perintah ini:
- Mengambil semua pixel dari canvas
- Mengubahnya menjadi format PNG
- Mengembalikan sebagai "data URL" (string panjang)

Contoh hasil:
```
data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAfQAAAH0CAAA...
```

#### STEP 2-3: Membuat Link Element

```javascript
const link = document.createElement('a');
link.download = 'clock.png';
```

Kita membuat element `<a>` (link) secara dinamis dengan:
- `download` attribute = nama file yang akan disimpan
- `href` attribute = URL file yang akan didownload

#### STEP 4-5: Trigger Download

```javascript
link.href = dataURL;
link.click();
```

Kita set `href` ke data URL, lalu "klik" linknya secara otomatis. Browser akan langsung download file!

### Hasil:

Ketika user klik tombol "Save As Image", file `clock.png` akan terdownload dengan warna yang sudah dipilih user!

<a name="bonus-localstorage"></a>
## âœ¨ Bonus: Menyimpan Warna ke Local Storage

**Masalah:** Ketika page direload, warna balik ke default.  
**Solusi:** Gunakan Local Storage untuk menyimpan pilihan warna.

### Apa itu Local Storage?

Local Storage adalah "memory" di browser yang bisa menyimpan data secara permanen (sampai user bersihkan cache).

### Kode untuk Menyimpan Warna:

```javascript
// Ketika user mengubah warna, simpan ke local storage
faceColor.addEventListener('change', () => {
  localStorage.setItem('faceColor', faceColor.value);
});

borderColor.addEventListener('change', () => {
  localStorage.setItem('borderColor', borderColor.value);
});

lineColor.addEventListener('change', () => {
  localStorage.setItem('lineColor', lineColor.value);
});

largeHandColor.addEventListener('change', () => {
  localStorage.setItem('largeHandColor', largeHandColor.value);
});

secondHandColor.addEventListener('change', () => {
  localStorage.setItem('secondHandColor', secondHandColor.value);
});
```

### Kode untuk Membaca Warna Tersimpan:

```javascript
// Ketika page dimuat, ambil warna dari local storage
window.addEventListener('load', () => {
  const savedFaceColor = localStorage.getItem('faceColor');
  const savedBorderColor = localStorage.getItem('borderColor');
  const savedLineColor = localStorage.getItem('lineColor');
  const savedLargeHandColor = localStorage.getItem('largeHandColor');
  const savedSecondHandColor = localStorage.getItem('secondHandColor');
  
  // Jika ada data tersimpan, gunakan itu (jika tidak, gunakan default)
  if (savedFaceColor) faceColor.value = savedFaceColor;
  if (savedBorderColor) borderColor.value = savedBorderColor;
  if (savedLineColor) lineColor.value = savedLineColor;
  if (savedLargeHandColor) largeHandColor.value = savedLargeHandColor;
  if (savedSecondHandColor) secondHandColor.value = savedSecondHandColor;
});
```

**ğŸ¯ Hasil:** Sekarang ketika user kembali ke halaman, warna pilihan mereka masih ada!

```javascript
// Ambil semua element
const canvas = document.getElementById('canvas');
const faceColor = document.getElementById('face-color');
const borderColor = document.getElementById('border-color');
const lineColor = document.getElementById('line-color');
const largeHandColor = document.getElementById('large-hand-color');
const secondHandColor = document.getElementById('second-hand-color');

// Fungsi menggambar jam
function clock() {
  const now = new Date();
  const ctx = canvas.getContext('2d');

  // Setup canvas
  ctx.save();
  ctx.clearRect(0, 0, 500, 500);
  ctx.translate(250, 250);
  ctx.rotate(-Math.PI / 2);

  // Warna default
  ctx.strokeStyle = '#000000';
  ctx.fillStyle = '#f4f4f4';
  ctx.lineWidth = 5;
  ctx.lineCap = 'round';

  // Draw clock face/border
  ctx.save();
  ctx.beginPath();
  ctx.lineWidth = 14;
  ctx.strokeStyle = borderColor.value;
  ctx.fillStyle = faceColor.value;
  ctx.arc(0, 0, 142, 0, Math.PI * 2, true);
  ctx.stroke();
  ctx.fill();
  ctx.restore();

  // Draw hour lines
  ctx.save();
  ctx.strokeStyle = lineColor.value;
  for (let i = 0; i < 12; i++) {
    ctx.beginPath();
    ctx.rotate(Math.PI / 6);
    ctx.moveTo(100, 0);
    ctx.lineTo(120, 0);
    ctx.stroke();
  }
  ctx.restore();

  // Draw minute lines
  ctx.save();
  ctx.strokeStyle = lineColor.value;
  ctx.lineWidth = 4;
  for (let i = 0; i < 60; i++) {
    if (i % 5 !== 0) {
      ctx.beginPath();
      ctx.moveTo(117, 0);
      ctx.lineTo(120, 0);
      ctx.stroke();
    }
    ctx.rotate(Math.PI / 30);
  }
  ctx.restore();

  // Get current time
  const hr = now.getHours() % 12;
  const min = now.getMinutes();
  const sec = now.getSeconds();

  // Draw hour hand
  ctx.save();
  ctx.rotate(
    (Math.PI / 6) * hr + (Math.PI / 360) * min + (Math.PI / 21600) * sec
  );
  ctx.strokeStyle = largeHandColor.value;
  ctx.lineWidth = 14;
  ctx.beginPath();
  ctx.moveTo(-20, 0);
  ctx.lineTo(80, 0);
  ctx.stroke();
  ctx.restore();

  // Draw minute hand
  ctx.save();
  ctx.rotate((Math.PI / 30) * min + (Math.PI / 1800) * sec);
  ctx.strokeStyle = largeHandColor.value;
  ctx.lineWidth = 10;
  ctx.beginPath();
  ctx.moveTo(-28, 0);
  ctx.lineTo(112, 0);
  ctx.stroke();
  ctx.restore();

  // Draw second hand
  ctx.save();
  ctx.rotate((sec * Math.PI) / 30);
  ctx.strokeStyle = secondHandColor.value;
  ctx.fillStyle = secondHandColor.value;
  ctx.lineWidth = 6;
  ctx.beginPath();
  ctx.moveTo(-30, 0);
  ctx.lineTo(100, 0);
  ctx.stroke();
  ctx.beginPath();
  ctx.arc(0, 0, 10, 0, Math.PI * 2, true);
  ctx.fill();
  ctx.restore();

  ctx.restore();

  requestAnimationFrame(clock);
}

// Mulai jam
requestAnimationFrame(clock);

// Event listener untuk save button
document.getElementById('save-btn').addEventListener('click', () => {
  const dataURL = canvas.toDataURL('image/png');
  const link = document.createElement('a');
  link.download = 'clock.png';
  link.href = dataURL;
  link.click();
});
```
